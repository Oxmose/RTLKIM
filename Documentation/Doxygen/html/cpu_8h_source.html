197 /* Return highest supported input value for cpuid instruction. ext can
198  * be either 0x0 or 0x80000000 to return highest supported value for
199  * basic or extended cpuid information. Function returns 0 if cpuid
200  * is not supported or whatever cpuid returns in eax register. If sig
201  * pointer is non-null, then first four bytes of the SIG
202  * (as found in ebx register) are returned in location pointed by sig.
203  *
204  * @param ext The opperation code for the CPUID instruction.
205  * @returns The highest supported input value for cpuid instruction.
206  */
207 __inline__ static uint32_t get_cpuid_max (uint32_t ext)
208 {
209  uint32_t regs[4];
210  if(cpuid_capable() == 0)
211  {
212  return 0;
213  }
214 
215  /* Host supports cpuid. Return highest supported cpuid input value. */
216  __asm__ __volatile__("cpuid":"=a"(*regs),"=b"(*(regs+1)),
217  "=c"(*(regs+2)),"=d"(*(regs+3)):"a"(ext));
218 
219  return regs[0];
220 }
221 
222 /* Return cpuid data for requested cpuid leaf, as found in returned
223  * eax, ebx, ecx and edx registers. The function checks if cpuid is
224  * supported and returns 1 for valid cpuid information or 0 for
225  * unsupported cpuid leaf. All pointers are required to be non-null.
226  *
227  * @param code The opperation code for the CPUID instruction.
228  * @param regs The register used to store the CPUID instruction return.
229  * @returns 1 in case of succes, 0 otherwise.
230  */
231 
232 __inline__ static int32_t cpuid (uint32_t code,
233  uint32_t regs[4])
234 {
235  if(cpuid_capable() == 0)
236  {
237  return 0;
238  }
239  uint32_t ext = code & 0x80000000;
240  uint32_t maxlevel = get_cpuid_max(ext);
241 
242  if (maxlevel == 0 || maxlevel < code)
243  {
244  return 0;
245  }
246  __asm__ __volatile__("cpuid":"=a"(*regs),"=b"(*(regs+1)),
247  "=c"(*(regs+2)),"=d"(*(regs+3)):"a"(code));
248  return 1;
249 }

369 /* Compare and swap word atomicaly.
370  *
371  * @returns The value of the lock

376 __inline__ static uint32_t cpu_compare_and_swap(volatile uint32_t* p_val,
377  int oldval, int newval)

379  uint8_t prev;
380  __asm__ __volatile__ (
